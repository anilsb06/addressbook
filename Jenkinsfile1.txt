pipeline {
    agent {
        docker { image 'maven' }
    }
    stages {
        stage('compile') {
            steps {
                sh 'mvn compile'
            }
        }
        stage('test') {
            steps {
                sh 'mvn test'
                testresults: '**TEST*.xml'
            }
        }
        stage('test') {
            steps {
                sh 'mvn package'
                warfile:  '**/target/addressbook.war'
            }
        }
    }
}
